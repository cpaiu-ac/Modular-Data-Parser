#include "header.h"
#include <stdio.h>
#include <stdlib.h>

void citeste_asociatie(Asociatie* asociatie, const char* fisier_intrare) {
    FILE* fisier = fopen(fisier_intrare, "r");
    if (fisier == NULL) {
        printf("Eroare la deschiderea fisierului %s\n", fisier_intrare);
        exit(1);
    }

    fscanf(fisier, "%u", &(asociatie->numar_apicultori));

    asociatie->apicultori = malloc(asociatie->numar_apicultori * sizeof(Apicultor));
    if (asociatie->apicultori == NULL) {
        printf("Eroare la alocarea memoriei\n");
        exit(1);
    }

    for (unsigned int i = 0; i < asociatie->numar_apicultori; i++) {
        Apicultor* apicultor = &(asociatie->apicultori[i]);

        fscanf(fisier, "%u", &(apicultor->numar_recolte));
        fscanf(fisier, "%u", &(apicultor->numar_stupi));

        apicultor->productie_miere = malloc(apicultor->numar_recolte * sizeof(float*));
        if (apicultor->productie_miere == NULL) {
            printf("Eroare la alocarea memoriei\n");
            exit(1);
        }

        for (unsigned int j = 0; j < apicultor->numar_recolte; j++) {
            apicultor->productie_miere[j] = malloc(apicultor->numar_stupi * sizeof(float));
            if (apicultor->productie_miere[j] == NULL) {
                printf("Eroare la alocarea memoriei\n");
                exit(1);
            }

            for (unsigned int k = 0; k < apicultor->numar_stupi; k++) {
                fscanf(fisier, "%f", &(apicultor->productie_miere[j][k]));
            }
        }
    }

    fclose(fisier);
}

void afiseaza_asociatie(const Asociatie* asociatie) {
    printf("Numarul de apicultori: %u\n\n", asociatie->numar_apicultori);

    for (unsigned int i = 0; i < asociatie->numar_apicultori; i++) {
        const Apicultor* apicultor = &(asociatie->apicultori[i]);

        printf("Apicultor %u\n", i + 1);
        printf("Numarul de recolte: %u\n", apicultor->numar_recolte);
        printf("Numarul de stupi: %u\n", apicultor->numar_stupi);
        printf("Productia de miere:\n");

        for (unsigned int j = 0; j < apicultor->numar_recolte; j++) {
            for (unsigned int k = 0; k < apicultor->numar_stupi; k++) {
                printf("%.2f ", apicultor->productie_miere[j][k]);
            }
            printf("\n");
        }

        printf("\n");
    }
}

void determina_apicultor_max_productie(const Asociatie* asociatie, unsigned int recolta) {
    if (recolta < 1 || recolta > asociatie->apicultori[0].numar_recolte) {
        printf("Recolta indicata nu exista.\n");
        return;
    }

    unsigned int apicultor_max = 0;
    float max_productie = asociatie->apicultori[0].productie_miere[recolta - 1][0];

    for (unsigned int i = 1; i < asociatie->numar_apicultori; i++) {
        float productie = asociatie->apicultori[i].productie_miere[recolta - 1][0];

        if (productie > max_productie) {
            max_productie = productie;
            apicultor_max = i;
        }
    }

    printf("Apicultorul cu cea mai mare cantitate de miere la recolta %u este apicultorul %u\n",
           recolta, apicultor_max + 1);
}

void elibereaza_memorie(Asociatie* asociatie) {
    for (unsigned int i = 0; i < asociatie->numar_apicultori; i++) {
        Apicultor* apicultor = &(asociatie->apicultori[i]);

        for (unsigned int j = 0; j < apicultor->numar_recolte; j++) {
            free(apicultor->productie_miere[j]);
        }

        free(apicultor->productie_miere);
    }

    free(asociatie->apicultori);
}
